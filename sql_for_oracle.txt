0 sno number primary key,
1 name varchar2(10),
2 tel varchar2(20) unique,
3 birth varchar2(20),
4 id varchar2(10) unique,
5 pw varchar2(10),

n_cash number,

bno number primary key,
name varchar2(10),
tel varchar2(20) unique,
birth varchar2(10),
pw varchar2(10),
point number,
lv number,
joindate varchar2(20));

pno number primary key,
name varchar2(20) unique,
company varchar2(10),
orderprice number,
sellprice number,
pqty number,
barcode varchar2(20) unique);

pno number,
pqty number,

bno number,
selldate varchar2(20));

create table buy_log(
buyno number primary key,
sno number,
pno number,
pqty number,
price number,
orderdate varchar2(20));
